	public void mainKabul() {
		while(true) {
			try {
				Thread.sleep(100);
			} 
			catch (InterruptedException e) {}
			synchronized(lock) {
				if(istekSayisi==istenilenIstekSayisi) {
					System.out.println("kabul edilen istek sayisi "+istekSayisi);
					System.out.println("cevaplanan istek sayisi "+cevaplananistekSayisi);
					System.out.println("toplam sunucu sayisi: "+yenisub+1);
					break;
				}
				if(kapasite==10000) {
					try {
						lock.wait();
					} catch (InterruptedException e) {
						// TODO Auto-generated catch block
						e.printStackTrace();
					}
				}
				int n = rand.nextInt(100);
				n += 1;
				if(kapasite+n<=10000 && istekSayisi+n<=istenilenIstekSayisi) {
					istekSayisi+=n;
					//System.out.println("main istek kabul "+n);
					lock2.lock();
					kapasite=kapasite+n;
					String yaz=String.valueOf(kapasite);
					anaKapasite.setText(yaz);
					//System.out.println("main kapasite "+kapasite);
					lock2.unlock();
				}
				else {
					if(istekSayisi+n>istenilenIstekSayisi) {
						//System.out.println("main istek kabul "+(istenilenIstekSayisi-istekSayisi));
						lock2.lock();
						kapasite=kapasite+istenilenIstekSayisi-istekSayisi;
						lock2.unlock();
						istekSayisi=istenilenIstekSayisi;
						String yaz=String.valueOf(kapasite);
						anaKapasite.setText(yaz);
						//System.out.println("main kapasite "+kapasite);
					}
					else {
						istekSayisi=istekSayisi+10000-kapasite;
						lock2.lock();
						//System.out.println("main istek kabul "+(10000-kapasite));
						kapasite=10000;
						String yaz=String.valueOf(kapasite);
						anaKapasite.setText(yaz);
						//System.out.println("main kapasite "+kapasite);
						lock2.unlock();
					}
				}System.out.println("main kapasite % "+(kapasite*100/10000));
				lock.notify();
			}
		}	
	}

					lock5.unlock();
					lock2.lock();
					kapasite-=n;
					lock2.unlock();
				}
				else {
					//System.out.println("main istek cevap "+kapasite);
					lock5.lock();
					cevaplananistekSayisi+=kapasite;
					mainCevaplanan+=kapasite;
					String yaz=String.valueOf(mainCevaplanan);
					anaCevaplanan.setText(yaz);
					lock5.unlock();
					lock2.lock();
					kapasite=0;
					lock2.unlock();

				anaKapasite.setText(yaz);
				System.out.println("main kapasite % "+(kapasite*100/10000));
				System.out.println("main cevaplanan = "+mainCevaplanan);
				//System.out.println("main kapasite "+kapasite);
				lock.notify();
			}
		}
